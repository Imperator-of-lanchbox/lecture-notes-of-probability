\section{Basic Logic for Ancient Greeks}
\begin{theorem}\label{thm:mortal}
All men are mortal.
\end{theorem}

\begin{fact}\label{fact:socrates}
Socrates is a man.
\end{fact}
%use \cref instead of \ref
Combine \cref{thm:mortal} and \cref{fact:socrates}
and we get 
\begin{corollary}
    Socrates is mortal.
\end{corollary}
